「mysql」学习笔记

why MySQL

MySQL 是最流行的关系型数据库管理系统，在 WEB 应用方面 MySQL 是最好的 RDBMS(Relational Database Management System：关系数据库管理系统)应用软件之一。

基本操作

  • 启动(macOS) - mysql.server start
  • 进入mysql命令行 - mysql -u root -p
  • 创建数据库 - create database 库名；
  • 查看数据库 - show databases;
  • 删除数据库 - drop database 库名;
  • 打开数据库 - use 库名;
  • 查看数据表 - show tables;
  • 查看数据表详情 - desc 表名;
  • 退出mysql命令行 - quit; / exit;

数据表

1. 数据表创建&删除

# 
create table stu (id int NOT NULL,name char(10) NOT NULL,age char(2),sex DEFAULT '',PRIMARY KEY (id));
# 
drop table stu;
# 
desc stu;

2. 查询

# 查询stu表的全部数据
select * from stu
# 查询2-5条
select * from stu limit 1，4
# 前三条
select * from stu limit 3
# 按条件查询
select id,name,age from stu where id=2;

3. 插入

#
insert into stu values(1,'cxf',23,'男');
# 
insert into stu (id,name,age,sex) values(1,'cxf',23,'男');

4. 修改

update stu set age=24 where name='';

5. 删除

delete from stu where id=1;
