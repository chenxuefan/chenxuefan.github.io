计算运行时间

import multiprocessing as mp
import time

def target(arg):
  ...

if __name__ == '__main__'
  start = time.time()

  processes = []
  for i in range(5):
    process = mp.Process(target=target, args=(i,))
    process.start() # 
    processes.append(process)
  for process in processes:
    process.join() # 

  end = time.time()
  print(f'共耗时：{end-start} s')
